-- Table: habitude.localisation_raw

-- DROP TABLE habitude.localisation_raw;

CREATE TABLE habitude.localisation_raw
(
  "userID" integer NOT NULL,
  date_time timestamp without time zone NOT NULL,
  longitude real,
  latitude real,
  CONSTRAINT pkey_localisation_raw PRIMARY KEY ("userID", date_time)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE habitude.localisation_raw OWNER TO postgres;



-- Table: habitude.geo_tempo_events

-- DROP TABLE habitude.geo_tempo_events;

CREATE TABLE habitude.geo_tempo_events
(
  "userID" integer NOT NULL,
  "timeIn" timestamp without time zone NOT NULL,
  "timeOut" timestamp without time zone,
  "zoneID" integer,
  interest character varying(254)[],
  CONSTRAINT pkey_events PRIMARY KEY ("userID", "timeIn")
)
WITH (
  OIDS=FALSE
);
ALTER TABLE habitude.geo_tempo_events OWNER TO postgres;

CREATE TABLE habitude.weekly_habit
(

WITH (
  OIDS=FALSE
);
